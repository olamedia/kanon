    public function set($value){
        if (is_array($value)){
            $this->_value = array();
            foreach ($value as $x){
                $x = strval($x);
                $this->_value[$x] = $x;
            }
        }else{
            $this->_value = $value;
        }
    }

    public function push($value){
        $value = strval($value);
        if (!is_array($this->_value)){
            $this->_value = array($this->_value=>$this->_value);
        }
        $this->_value[$value] = $value;
    }

    public function pop($value){
        $value = strval($value);
        if (!is_array($this->_value)){
            $this->_value = array($this->_value=>$this->_value);
        }
        unset($this->_value[$value]);
    }
